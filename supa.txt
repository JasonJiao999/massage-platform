1-add_contribution_points

DECLARE
    current_daily_points INT;
    last_gain DATE;
    points_to_apply INT;
    daily_cap INT := 50; -- 在这里设置每日积分上限
BEGIN
    -- 1. 获取用户当前的每日积分缓存和最后更新日期
    SELECT daily_points_cache, last_point_gain_at::date INTO current_daily_points, last_gain
    FROM public.profiles WHERE id = target_user_id;

    -- 2. 检查是否是新的一天，如果是，则重置每日积分缓存
    IF last_gain IS NULL OR last_gain < CURRENT_DATE THEN
        current_daily_points := 0;
    END IF;

    -- 3. 计算本次实际可以增加的积分
    points_to_apply := LEAST(points_to_add, daily_cap - current_daily_points);

    -- 4. 如果有积分可以增加，则执行操作
    IF points_to_apply > 0 THEN
        -- a. 记录积分变动日志 (事实来源)
        INSERT INTO public.credit_logs (user_id, points_change, reason_code, description, related_booking_id)
        VALUES (target_user_id, points_to_apply, reason, details, booking_ref_id);

        -- b. 更新 profiles 表中的总分和每日缓存
        UPDATE public.profiles
        SET
            contribution_score = contribution_score + points_to_apply,
            daily_points_cache = current_daily_points + points_to_apply,
            last_point_gain_at = now()
        WHERE id = target_user_id;
    END IF;

END;

2-add_points_and_level_up

DECLARE
  current_points INT;
  new_points INT;
  current_level INT;
  new_level INT;
BEGIN
  -- 獲取用戶當前的推薦積分和等級
  SELECT referral_points, level INTO current_points, current_level
  FROM public.profiles
  WHERE id = user_id;

  -- 計算新積分
  new_points := COALESCE(current_points, 0) + points_to_add;
  
  -- **核心邏輯**：根據總積分計算新等級 (例如: 210 積分 = 2 級)
  new_level := new_points / 100; 

  -- 更新用戶的積分和等級
  UPDATE public.profiles
  SET 
    referral_points = new_points,
    level = new_level
  WHERE id = user_id;
END;

3-handle_level_up

DECLARE
  new_level INT;
BEGIN
  -- 根据我们设计的“游戏规则”来判断新等级
  -- 您可以随时在这里修改等级和分数阈值
  IF NEW.contribution_score >= 1500 THEN
    new_level := 5;
  ELSIF NEW.contribution_score >= 700 THEN
    new_level := 4;
  ELSIF NEW.contribution_score >= 300 THEN
    new_level := 3;
  ELSIF NEW.contribution_score >= 100 THEN
    new_level := 2;
  ELSE
    new_level := 1;
  END IF;

  -- 只有当计算出的新等级高于当前等级时，才执行更新
  IF new_level > OLD.level THEN
    NEW.level := new_level;
  END IF;

  -- 返回修改后的行数据，让 UPDATE 操作继续
  RETURN NEW;
END;

4-handle_new_user

DECLARE
  referrer_id UUID;
  points_for_referrer INT := 50; -- ** 您可以修改：推薦人獲得的積分
  points_for_referee INT := 25; -- ** 您可以修改：新用戶獲得的積分
  used_code TEXT;
BEGIN
  -- 1. 從傳入的 'raw_user_meta_data' 中獲取推薦碼
  used_code := new.raw_user_meta_data->>'referral_code';

  -- 2. 創建新的 'profiles' 記錄
  insert into public.profiles (
    id, 
    email, 
    role, 
    nickname, 
    full_name,
    referral_code --
  )
  values (
    new.id, 
    new.email, 
    'customer', -- 默認角色
    new.raw_user_meta_data->>'nickname',
    new.raw_user_meta_data->>'full_name',
    generate_referral_code() -- 為新用戶生成他們自己的推薦碼
  );

  -- 3. 如果新用戶在註冊時使用了有效的推薦碼
  IF used_code IS NOT NULL THEN
    -- 根據推薦碼查找推薦人
    SELECT id INTO referrer_id FROM public.profiles WHERE referral_code = used_code;
    
    IF referrer_id IS NOT NULL THEN
      -- A. 為推薦人 (referrer) 添加積分
      PERFORM add_points_and_level_up(referrer_id, points_for_referrer);
      
      -- B. 為新用戶 (referee) 添加積分
      PERFORM add_points_and_level_up(new.id, points_for_referee);
      
      -- C. 記錄是誰推薦了他們
      UPDATE public.profiles SET referred_by = referrer_id WHERE id = new.id;
    END IF;
  END IF;

  return new;
end;
